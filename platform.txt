name=ChipIntelli CI130X Arduino
version=1.0.0

# CI13XX SDK ASR ALG V2.7.12 is ABI-locked to the vendor Nuclei GCC 9.2.0
# distribution. Boards Manager removes the archive's single gcc_fix_raissrc/
# wrapper directory, so the installed tool root contains bin/ directly.
compiler.path={runtime.tools.riscv-gcc.path}/bin/
compiler.prefix=riscv-nuclei-elf-

compiler.c.cmd={compiler.prefix}gcc
compiler.cpp.cmd={compiler.prefix}g++
compiler.S.cmd={compiler.prefix}gcc
compiler.c.elf.cmd={compiler.prefix}g++
compiler.ar.cmd={compiler.prefix}gcc-ar
compiler.objcopy.cmd={compiler.prefix}objcopy
compiler.size.cmd={compiler.prefix}size

compiler.common.flags=-march=rv32imafc -mabi=ilp32f -mcmodel=medlow -msmall-data-limit=8 -msave-restore -mfdiv -Os -flto -fsigned-char -ffunction-sections -fdata-sections -fno-common -fno-delete-null-pointer-checks -fno-unroll-loops -fshort-enums
compiler.sdk.defines=-DASR_CODE_VERSION=2 -DCORE_ID=0
compiler.sdk.includes="-I{build.variant.path}" "-I{runtime.platform.path}/tools/sdk/include" "-I{runtime.platform.path}/tools/sdk/include/sdk"

compiler.c.flags={compiler.common.flags} -g -std=gnu11 -MMD
compiler.cpp.flags={compiler.common.flags} -g -std=gnu++17 -MMD -fno-exceptions -fno-rtti -fno-use-cxa-atexit -fno-threadsafe-statics
compiler.S.flags={compiler.common.flags} -g -x assembler-with-cpp -MMD
compiler.ar.flags=crs

compiler.c.extra_flags=
compiler.cpp.extra_flags=
compiler.S.extra_flags=
compiler.c.elf.extra_flags=
compiler.ar.extra_flags=
compiler.objcopy.extra_flags=
compiler.libraries.ldflags=

compiler.c.elf.flags={compiler.common.flags} -g -nostartfiles --specs=nano.specs -Wl,--gc-sections -Wl,-u,_start -Wl,-u,main "-Wl,--Map={build.path}/{build.project_name}.map" "-L{runtime.platform.path}/tools/sdk/lib/{build.variant}" "-L{runtime.platform.path}/tools/sdk/lib" "-L{runtime.platform.path}/tools/sdk/ld" "-T{runtime.platform.path}/tools/sdk/ld/ci130x_alg_pro_null.lds"
compiler.c.elf.wrap=-Wl,--wrap,memcmp -Wl,--wrap,memcpy -Wl,--wrap,memmove -Wl,--wrap,memset -Wl,--wrap,strcat -Wl,--wrap,strcmp -Wl,--wrap,strcpy -Wl,--wrap,strlen -Wl,--wrap,strncat -Wl,--wrap,strncmp -Wl,--wrap,strncpy -Wl,--wrap,strstr -Wl,--wrap,stpcpy -Wl,--wrap,stpncpy -Wl,--wrap,strnlen -Wl,--wrap,strtoul -Wl,--wrap,atol -Wl,--wrap,qsort -Wl,--wrap,malloc -Wl,--wrap,free -Wl,--wrap,vsnprintf -Wl,--wrap,sprintf -Wl,--wrap,sscanf -Wl,--wrap,abs -Wl,--wrap,isnan -Wl,--wrap,isinf -Wl,--wrap,cosf -Wl,--wrap,sinf -Wl,--wrap,tanf -Wl,--wrap,expf -Wl,--wrap,sqrtf -Wl,--wrap,fabsf -Wl,--wrap,logf -Wl,--wrap,log10f -Wl,--wrap,atoi -Wl,--wrap,vTaskStartScheduler -Wl,--wrap,deal_asr_msg_by_cmd_id -Wl,--wrap,sys_asr_result_hook -Wl,--wrap,ADC_IRQHandler
compiler.c.elf.libs=-lci13xx_sdk -lasr_dis_deepse_v2 -lir_data -lnlp -lnewlib_port -lfreertos_port -ldsu -lflash_encrypt -lcwsl_v2 -ltts -lOnMicroBLE -lcias_crypto -lcikd_pro_cwsl_dis_frm -lm

recipe.hooks.prebuild.1.pattern=powershell.exe -NoProfile -ExecutionPolicy Bypass -File "{runtime.platform.path}/tools/prepare_resources.ps1" -ProjectPath "{build.source.path}" -PackageResources "{runtime.platform.path}/recursos"

recipe.c.o.pattern="{compiler.path}{compiler.c.cmd}" {compiler.c.flags} {compiler.c.extra_flags} -DF_CPU={build.f_cpu} -DARDUINO={runtime.ide.version} -DARDUINO_{build.board} -DARDUINO_ARCH_CI13XX {compiler.sdk.defines} {build.extra_flags} {compiler.sdk.includes} {includes} -c "{source_file}" -o "{object_file}"
recipe.cpp.o.pattern="{compiler.path}{compiler.cpp.cmd}" {compiler.cpp.flags} {compiler.cpp.extra_flags} -DF_CPU={build.f_cpu} -DARDUINO={runtime.ide.version} -DARDUINO_{build.board} -DARDUINO_ARCH_CI13XX {compiler.sdk.defines} {build.extra_flags} {compiler.sdk.includes} {includes} -c "{source_file}" -o "{object_file}"
recipe.S.o.pattern="{compiler.path}{compiler.S.cmd}" {compiler.S.flags} {compiler.S.extra_flags} -DF_CPU={build.f_cpu} -DARDUINO={runtime.ide.version} -DARDUINO_{build.board} -DARDUINO_ARCH_CI13XX {compiler.sdk.defines} {build.extra_flags} {compiler.sdk.includes} {includes} -c "{source_file}" -o "{object_file}"
recipe.ar.pattern="{compiler.path}{compiler.ar.cmd}" {compiler.ar.flags} {compiler.ar.extra_flags} "{archive_file_path}" "{object_file}"

# The vendor startup, SDK main(), FreeRTOS, ASR tasks and board initialization are
# retained inside libci13xx_sdk.a. The Arduino core wraps scheduler startup to add
# setup()/loop() without replacing the validated SDK boot sequence.
recipe.c.combine.pattern="{compiler.path}{compiler.c.elf.cmd}" {compiler.c.elf.flags} {compiler.c.elf.wrap} {compiler.c.elf.extra_flags} -o "{build.path}/{build.project_name}.elf" -Wl,--start-group {object_files} "{archive_file_path}" {compiler.libraries.ldflags} {compiler.c.elf.libs} -Wl,--end-group -Wl,-EL

# First build the CI13XX two-image user-code container, then compose and inspect
# the complete firmware that the upload recipe writes from Flash address 0.
recipe.objcopy.bin.pattern=powershell.exe -NoProfile -ExecutionPolicy Bypass -File "{runtime.platform.path}/tools/postbuild.ps1" -Elf "{build.path}/{build.project_name}.elf" -Output "{build.path}/{build.project_name}.user_code.bin" -FirmwareOutput "{build.path}/{build.project_name}.bin" -Objcopy "{compiler.path}{compiler.objcopy.cmd}" -PlatformPath "{runtime.platform.path}" -CitoolCli "{runtime.tools.citool-cli.path}/citool-cli.exe" -ProjectResources "{build.source.path}/recursos" -Chip "{build.chip}" -MaxUserCodeSize {build.user_code_max}

recipe.output.tmp_file={build.project_name}.bin
recipe.output.save_file={build.project_name}.{build.variant}.firmware.bin

recipe.size.pattern="{compiler.path}{compiler.size.cmd}" -A "{build.path}/{build.project_name}.elf"
recipe.size.regex=^(?:\.boot_parameter|\.init|\.ilalign|\.text|\.ro_data|\.data|\.preinit_array|\.init_array|\.fini_array|\.ctors|\.dtors|\.gcc_except_table\S*)\s+([0-9]+).*
recipe.size.regex.data=^(?:\.data|\.bss|\.no_init|\.stack|\.sys_heap)\s+([0-9]+).*

tools.citool-cli.path={runtime.tools.citool-cli.path}
tools.citool-cli.cmd=citool-cli
tools.citool-cli.cmd.windows=citool-cli.exe
tools.citool-cli.upload.params.verbose=
tools.citool-cli.upload.params.quiet=
tools.citool-cli.upload.pattern="{path}/{cmd}" flash --port "{serial.port}" "{build.path}/{build.project_name}.bin"
tools.citool-cli.program.params.verbose=
tools.citool-cli.program.params.quiet=
tools.citool-cli.program.pattern="{path}/{cmd}" flash --port "{serial.port}" "{build.path}/{build.project_name}.bin"
